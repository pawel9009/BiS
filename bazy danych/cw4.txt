
DECLARE
     v_Record  countries%ROWTYPE;
BEGIN
select country_name, region_id into v_Record
from countries
where country_id ='CA';
 DBMS_OUTPUT.PUT_LINE(v_Record);
END;

DECLARE
TYPE DEPARTMENT_TYPE IS  TABLE OF NUMBER INDEX BY VARCHAR(64);
DEPARTAMENTY DEPARTMENT_TYPE;
BEGIN
FOR x in 1..10
LOOP
   SELECT DEPARTMENT_NAME INTO DEPARTAMENTY(x)
    FROM DEPARTMENTS
    WHERE DEPARTMENT_ID = x*10;
    DBMS_OUTPUT.PUT_LINE(DEPARTAMENTY(x));
END LOOP;
END;


DECLARE
TYPE DEPARTMENT_TYPE IS  TABLE OF DEPARTMENTS%ROWTYPE INDEX BY VARCHAR(64);
DEPARTAMENTY DEPARTMENT_TYPE;
BEGIN
FOR x in 1..10
LOOP
   SELECT DEPARTMENT_NAME INTO DEPARTAMENTY(x)
    FROM DEPARTMENTS
    WHERE DEPARTMENT_ID = x*10;
    DBMS_OUTPUT.PUT_LINE(DEPARTAMENTY(x).DEPARTMENT_ID||' '|| DEPARTAMENTY(x).DEPARTMENT_NAME
    ||" "|| DEPARTAMENTY(x).MANAGER_ID||' ' ||DEPARTAMENTY(x).LOCATION_ID);
END LOOP;
END;
 
 
 DECLARE
   CURSOR SALARY_LN IS
     SELECT SALARY, LAST_NAME FROM EMPLOYEES WHERE DEPARTMENT_ID= 50;
   WYNIK SALARY_LN%ROWTYPE;
BEGIN
    OPEN SALARY_LN;
    LOOP
        FETCH SALARY_LN INTO WYNIK;
        EXIT WHEN SALARY_LN%NOTFOUND;
        IF WYNIK.SALARY < 3100 THEN
            
            DBMS_OUTPUT.PUT_LINE(WYNIK.last_name || ' - dać podwyżkę');
        ELSE
            DBMS_OUTPUT.PUT_LINE(WYNIK.last_name || ' - nie dawać podwyżki');
        END IF;
    END LOOP;
    CLOSE SALARY_LN;
END;
 
 
DECLARE
    CURSOR WYNIK(ZAROBKI_MIN NUMBER, ZAROBKI_MAX NUMBER, IMIE VARCHAR) IS
        SELECT SALARY, FIRST_NAME, LAST_NAME FROM EMPLOYEES
        WHERE (SALARY BETWEEN ZAROBKI_MIN AND ZAROBKI_MAX);
    DANE WYNIK%ROWTYPE;
BEGIN
    FOR DANE IN WYNIK(1000, 5000, 'A')
    LOOP
        DBMS_OUTPUT.PUT_LINE(DANE.SALARY || ' ' ||
            DANE.FIRST_NAME || ' ' || DANE.LAST_NAME);
    END LOOP;
    FOR DANE IN WYNIK(5000, 20000, 'U')
    LOOP
        DBMS_OUTPUT.PUT_LINE(DANE.salary || ' ' ||
            DANE.first_name || ' ' || DANE.last_name);
    END LOOP;
END;
